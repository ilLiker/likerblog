## 题目一

```
给定一个单链表的头节点head，请判断该链表是否为回文结构
栈方法特别简单（笔试用）
改元链表的方法就需要注意边界了（面试用）
```

```java
 /**
     * 方案一
     * 使用栈，把数据都push到栈中
     * 然后按照原链表的正序遍历，逐一与栈中的pop出的数据对比
     */
  // 栈方法
    // need n extra space
    public  static boolean isPalindrome1(Node head){
        Stack<Node> stack = new Stack<>();
        Node cur = head;
        while (cur != null){
            stack.push(cur);
            cur = cur.next;
        }
        while (head != null){
            if(head.value != stack.pop().value){
                return false;
            }
            head = head.next;
        }
        return true;
    }
```

```java
 /**
```
```java
     * 方案三
     * 将后半部分逆转，然后从两端取next逐一对比
     * 最后再把链表还原
     */
    // need O(1) extra space
    public static boolean isPalindrome3(Node head){
        if(head == null || head.next == null){
            return true;
        }
        // 奇数时获取中点，偶数时获取前中点
        // 奇数时 slow = n; fast = 2n - 1;
        // 偶数时 slow = n; fast = 2n
        Node n1 = head;
        Node n2 = head;
        while (n2.next != null && n2.next.next != null){
            n1 = n1.next;
            n2 = n2.next.next;
        }
        // 把n1的next节点即右半部分的第一个节点 存起来，
        // 由于n2这个变量已经没用了所以可以使用n2来存储
        n2 = n1.next;
        n1.next = null;
        Node n3;
        while (n2 != null){
            n3 = n2.next;
            n2.next = n1;
            n1=n2;
            n2=n3;
        }
        // head要存起来，这时n2和n1都为最后一个节点，可以使用n2来存储
        // n3为null可以用来存储最后一个节点 因为要在最后逆转链表
        n3 = n1;
        n2 = head;
        boolean res = true;
        while (n1 != null && n2 != null){
            if(n1.value != n2.value){
                res = false;
                break;
            }
            n1 = n1.next;
            n2 = n2.next;
        }
        // 再把链表复原
        n1 = n3.next;
        n3.next = null;
        while (n1 != null){
            n2 = n1.next;
            n1.next = n3;
            n3 = n1;
            n1 = n2;
        }
        return res;
    }
```

## 题目二

```
将单向链表按某值划分成小、中间相等、右边大的形式
把链表放入数组里，在数组上做partition（笔试用）
分成小、中、大三部分，再把各个部分之间串起来（面试用）
```

```java
 /**
     * 第一种方案
     * 把链表放入数组里，在数组上做partition
     * 需要额外空间复杂度O(n)
     */
    public static Node listPartition1(Node head,int pivot){
        if(head == null){
            return head;
        }
        Node cur = head;
        int i = 0;
        while (cur != null){
            i++;
            cur = cur.next;
        }
        Node[] nodeArr = new Node[i];
        cur = head;
        for (i = 0; i < nodeArr.length; i++) {
            nodeArr[i] = cur.value;
            cur = cur.next;
        }
        arrPartition(nodeArr,pivot);
        for (i = 1; i != nodeArr.length; i++) {
            nodeArr[i-1].next = nodeArr[i];
        }
        nodeArr[i-1].next = null;
        return nodeArr[0];
    }

    public static void arrPartition(Node[] nodeArr,int pivot){
        int small = -1;
        int big = nodeArr.length;
        int index = 0;
        while (index != big){
            if(nodeArr[index].value < pivot){
                swap(nodeArr,++small,index++);
            }else if(nodeArr[index].value == pivot){
                index++;
            }else {
                swap(nodeArr,--big,index);
            }
        }
    }

    public static void swap(Node[] nodeArr,int a,int b){
        Node tmp = nodeArr[a];
        nodeArr[a] = nodeArr[b];
        nodeArr[b] = tmp;
    }

```

```java
/**
     * 将小、中、大分为三个队列，然后再把三个链表串联起来
     * 额外复杂度O(1)
     */
    public static Node listPartition2(Node head,int pivot){
        Node sH = null;
        Node sT = null;
        Node eH = null;
        Node eT = null;
        Node mH = null;
        Node mT = null;
        Node next = null;
        while (head!=null){
            next = head.next;
            head.next = null;
            if(head.value < pivot){
                if(sH==null){
                    sH = head;
                    sT = head;
                }else {
                    sT.next = head;
                    sT = head;
                }
            }else if(head.value == pivot){
                if(eH == null){
                    eH = head;
                    eT = head;
                }else {
                    eT.next = head;
                    eT = head;
                }
            }else {
                if(mH == null){
                    mH = head;
                    mT = head;
                }else {
                    mT.next = head;
                    mT = head;
                }
            }
            head = next;
        }
        if(sT != null){
            sT.next = eH;
            eT = eT == null ? sT : eT;
        }
        if(eT != null){
            eT.next = mH;
        }
        return sH != null ? sH : (eH != null ? eH:mH);
    }
```

```java
 /**
     * 第一种方案
     * 创建一个Map，存储每个节点和每个节点的拷贝节点
     * 最后遍历map，把没个节点的拷贝节点的next和rand按照map中的k进行复制
     */
    public static Node copyListWithRand1(Node head){
        HashMap<Node,Node> map = new HashMap<>();
        Node cur = head;
        while (cur != null){
            map.put(cur,new Node(cur.value));
            cur = cur.next;
        }
        cur = head;
        while (cur != null){
            // cur lao
            // map.get(cur) 新
            map.get(cur).next = map.get(cur.next);
            map.get(cur).rand = map.get(cur.rand);
            cur = cur.next;
        }
        return map.get(head);
    }
```

```java
            cur.next = new Node(cur.value);

            cur = next;

            next = cur.next.next;
            curCopy = cur.next;

            cur = next;
        }

    }
```

```java
    public static Node noLoop(Node head1,Node head2){
        if(head1 == null || head2 == null){
            return null;
        }
        Node cur1 = head1;
        Node cur2 = head2;
        int n = 0;
        while (cur1.next != null){
            n ++;
            cur1 = cur1.next;
        }

        while (cur2.next != null){
            n--;
            cur2 = cur2.next;
        }
        if(cur1 != cur2){
            return null;
        }
        // n 链表1长度减去链表2长度的值
        cur1 = n > 0 ? head1:head2;
        cur2 = cur1 == head1 ? head2 : head1;
        n = Math.abs(n);
        while (n != 0){
            n --;
            cur1=cur1.next;
        }
        while (cur1 == cur2){
            cur1 = cur1.next;
            cur2 = cur2.next;
        }
        return cur1;
    }
```
